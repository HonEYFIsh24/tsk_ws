866 // Frame type identification implementation.
867 template<typename T>
868 bool Frame::Is() const {
869  switch (this->GetType()) {
870  case kDtFrameIrLeft:
871  case kDtFrameIrRight:
872  case kDtFrameIr:
873  return (typeid(T) == typeid(IrFrame) || typeid(T) == typeid(VideoFrame));
874  case kDtFrameDepth:
875  return (typeid(T) == typeid(DepthFrame) || typeid(T) == typeid(VideoFrame));
876  case kDtFrameColor:
877  case kDtFrameColorLeft:
878  case kDtFrameColorRight:
879  return (typeid(T) == typeid(ColorFrame) || typeid(T) == typeid(VideoFrame));
880  case kDtFrameGyro:
881  return (typeid(T) == typeid(GyroFrame));
882  case kDtFrameAccel:
883  return (typeid(T) == typeid(AccelFrame));
884  case kDtFramePoints:
885  return (typeid(T) == typeid(PointsFrame));
886  case kDtFrameRawPhase:
887  return (typeid(T) == typeid(RawPhaseFrame) || typeid(T) == typeid(VideoFrame));
888  case kDtFrameSet:
889  return (typeid(T) == typeid(FrameSet));
890  case kDtFramePalm:
891  return (typeid(T) == typeid(PalmFrame));
892  default:
893  std::cout << "dt::Frame::is() did not catch frame type: " << (int) this->GetType()
894  << std::endl;
895  break;
896  }
897  return false;
898 }
