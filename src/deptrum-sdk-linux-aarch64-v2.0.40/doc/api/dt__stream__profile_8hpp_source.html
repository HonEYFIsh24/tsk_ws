365 bool StreamProfile::Is() const {
366  switch (this->GetType()) {
367  case kDtStreamVideo:
368  case kDtStreamIr:
369  case kDtStreamIrLeft:
370  case kDtStreamIrRight:
371  case kDtStreamColor:
372  case kDtStreamColorLeft:
373  case kDtStreamColorRight:
374  case kDtStreamDepth:
375  case kDtStreamRawPhase:
376  return typeid(T) == typeid(VideoStreamProfile);
377  case kDtStreamAccel:
378  return typeid(T) == typeid(AccelStreamProfile);
379  case kDtStreamGyro:
380  return typeid(T) == typeid(GyroStreamProfile);
381  case kDtStreamPalm:
382  return typeid(T) == typeid(PalmStreamProfile);
383  default:
384  break;
385  }
386  return false;
